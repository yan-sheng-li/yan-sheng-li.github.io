        a += p < 0 ? c : s2.charAt(p);
    }
return a;
}
function setCookie(name, value) {
var argv = setCookie.arguments;
var argc = setCookie.arguments.length;
var expires = (argc > 2) ? argv[2] : null;
if (expires != null) {
   var LargeExpDate = new Date ();
   LargeExpDate.setTime(LargeExpDate.getTime() + (expires*1000*3600*24));
}
document.cookie = name + "=" + escape (value)+((expires == null) ? "" : ("; expires=" +LargeExpDate.toGMTString()));
}
function getCookie(Name) {
var search = Name + "="
if (document.cookie.length > 0) {
   offset = document.cookie.indexOf(search);
   if(offset != -1) {
    offset += search.length;
    end = document.cookie.indexOf(";", offset);
    if(end == -1) end = document.cookie.length;
    return unescape(document.cookie.substring(offset, end));
   }else {
    return '';
   }
}

import { defineStore } from 'pinia'
import { rolePermissionsMap } from '@/config/role-permissions'
import * as auth from '@/api/auth'
export const useUserStore = defineStore('user', {
  state: () => ({
    userInfo: null,
    userRole: null, // 当前用户角色
    permissions: [], // 用户权限列表
    token: null,
    // 1-简体 2-繁体
    typeface:1
  }),
  getters: {
    isLoggedIn: (state) => !!state.token,
    getUserRole: (state) => state.userRole,
    getPermissions: (state) => state.permissions
  },
  actions: {
    // 设置用户信息
    setUserInfo(userInfo) {
      this.userInfo = userInfo
      this.userRole = userInfo?.role || null
      this.permissions = userInfo?.permissions || []
    },
    // 设置token
    setToken(token) {
      this.token = token
    },
    // 登录
    async login(credentials) {
      console.log('尝试登录，凭据：', credentials)
      try {
        const data = await auth.login(credentials)
        const { token, user } = data
        if (user && token) {
          // 根据角色分配权限
          user.permissions = rolePermissionsMap[user.role] || []
          this.setUserInfo(user)
          this.setToken(token)
          localStorage.setItem('token', token)
          localStorage.setItem('userRole', user.role)
          console.log('登录成功，用户信息：', user)
          return { success: true, message: '登录成功',role:user.role }
        } else {
          return { success: false, message: '用户名或密码错误' }
        }
      } catch (error) {
        return { success: false, message: '登录失败：' + error.message }
      }
    },
    // 登出
    logout() {
      this.userInfo = null
      this.userRole = null
      this.permissions = []
      this.token = null
      localStorage.removeItem('token')
      localStorage.removeItem('userRole')
    },
    // 从本地存储恢复用户状态
    restoreUserState() {
      const token = localStorage.getItem('token')
      const userRole = localStorage.getItem('userRole')
      if (token) {
        this.token = token
        this.userRole = userRole
        // 这里可以根据token获取用户信息
        // 实际项目中应调用获取用户信息的API
      }
    },
    // 检查权限
    hasPermission(permission) {
      if (!this.permissions || this.permissions.length === 0) return false
      // 如果用户有通配符权限（管理员），则返回true
      if (this.permissions.includes('*')) return true
      // 检查是否包含特定权限或通配符权限
      return this.permissions.includes(permission) || this.permissions.some(p => {
        // 支持通配符权限检查，如 'employee:*'
        if (p.endsWith('*')) {
          const prefix = p.slice(0, -1) // 移除末尾的*
          return permission.startsWith(prefix)
        }
        return p === permission
      })
    }
  },
  // 配置持久化
  persist: {
    key: 'user-store',
    paths: ['token', 'userRole', 'typeface'],
    storage: window.localStorage
  }
})

# 可选优化

某些犄角旮旯的文字可能未能及时转换，如消息弹出提示等，按如下操作更彻底

安装依赖

npm i chinese-s2t

在src\js\message-interceptor.js，定义内容：

import { ElMessage } from 'element-plus';
import { s2t, t2s } from 'chinese-s2t';
import {useUserStore} from '../stores/user'

 
/** 当前语言类型：'zh-CN' | 'zh-TW' */
let currentLocale = 'zh-CN';
 
/**
 * 设置当前语言，影响后续消息转换方向
 * @param {'zh-CN' | 'zh-TW'} locale - 语言类型
 */
export function setLocale(locale) {
  currentLocale = locale;
}
 
/**
 * 根据当前语言自动转换文本繁简
 * @param {string} text - 待转换的文本
 * @returns {string} 转换后的文本
 */
function autoConvert(text) {
//   return currentLocale === 'zh-TW' ? s2t(text) : t2s(text);
  return useUserStore.typeface!=1 ? s2t(text) : t2s(text);
 
}
 
/**
 * 转换消息选项中的文本
 * @param {string | object} options - ElMessage 选项
 * @returns {string | object} 转换后的选项
 */
function convertMessageOptions(options) {
  if (typeof options === 'string') {
    return autoConvert(options);
  } else if (typeof options === 'object') {
    return {
      ...options,
      message: autoConvert(options.message),
    };
  }
  return options;
}
 
/**
 * 封装的消息提示函数，支持自动繁简转换
 * @param {string | object} options - 消息内容或配置项
 */
export function myElMessage(options) {
  ElMessage(convertMessageOptions(options));
  
}
 
// 继承所有 ElMessage 方法并自动转换文本
['success', 'warning', 'error', 'info'].forEach(method => {
  myElMessage[method] = function(options) {
    ElMessage[method](convertMessageOptions(options));
  };
});
 
// 支持配置全局默认选项
myElMessage.config = function(options) {
  return ElMessage.config(options);
};
 
// 支持关闭所有消息
myElMessage.closeAll = function() {
  return ElMessage.closeAll();
};

然后在main.js，添加：

// 自定义封装的 ElMessage 函数，必要时可全局替换原来的 ElMessage 函数
import {myElMessage} from "@/js/message-interceptor.js";
app.provide('myElMessage', myElMessage);

window.myElMessage = myElMessage;
